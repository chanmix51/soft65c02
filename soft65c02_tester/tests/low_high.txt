marker $$test symbol byte references$$

// Add a test symbol
symbols add test_str1=0x1234

// Test low and high byte comparisons with memory locations
memory write #0x2000 0x(34,12) // Write little endian value

// Assert that memory contains the correct bytes from symbol address
assert #0x2000 = <$test_str1  $$memory contains low byte of test_str1$$
assert #0x2001 = >$test_str1  $$memory contains high byte of test_str1$$

// Test with registers
registers set A=0x34 // Set A to low byte of 0x1234
registers set X=0x12 // Set X to high byte of 0x1234

assert A = <$test_str1  $$A register contains low byte of test_str1$$
assert X = >$test_str1  $$X register contains high byte of test_str1$$
